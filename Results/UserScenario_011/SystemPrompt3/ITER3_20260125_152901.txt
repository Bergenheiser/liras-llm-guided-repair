LabSampleDelivery

agents:
	humans: Nurse, LabPersonnel
	robots: RobotArm, SecureDrawerRobot
locations: patientFloor1, patientFloor2, laboratory, analysisMachine
resources: sample, barcode, lockedDrawer

Nurse:
	1: collect(sample)
	2: insertDataInTrackSystem(sample)
	3: requestDelivery(laboratory)

SecureDrawerRobot:
    1: moveTo patientFloor1
    2: receive(sample, barcode)
    3: moveTo laboratory
    4: deliveryTo(RobotArm)

RobotArm:
	1: moveTo laboratory
	2: pickSample(sample) from(lockedDrawer)
	3: scanBarcode(sample)
	4: sortAndLoad(sample, analysisMachine)
	5: logSamplePickInfo()
	6: trackTimeBetweenCollectionDelivery()

LabPersonnel:
	1: pickupFrom(lockedDrawer)
	2: scanBarcode(sample)
	3: sortAndLoad(sample, analysisMachine)
	4: logSamplePickInfo()
	5: trackTimeBetweenCollectionDelivery()
