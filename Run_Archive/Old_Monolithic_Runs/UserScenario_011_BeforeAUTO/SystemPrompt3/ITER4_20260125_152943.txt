LabSampleTransport

agents:
	humans: Nurse, LabPersonnel
	robots: TransportRobot, RoboticArm
locations: patientFloor, laboratory, analysisMachine
resources: sample, lockedDrawer

Nurse:
	1: collect(sample)
	2: insertSampleData(trackSystem)
	3: requestDelivery(TransportRobot)
TransportRobot:
	1: moveTo patientFloor
	2: load(sample, lockedDrawer)
	3: moveTo laboratory
	4: deliver(sample, LabPersonnel or RoboticArm)
RoboticArm:
	1:	stop 
	restart if deliveryAvailable(sample, lockedDrawer)
	2: pickUp(sample, lockedDrawer)
	3: scanBarcode(sample)
	4: sort(sample)
	5: load(sample, analysisMachine)
LabPersonnel:
	1: stop
	restart if deliveryAvailable(sample, lockedDrawer)
	2: open(lockedDrawer)
	3: receive(sample)
	4:	moveTo(analysisMachine)

Conclusion:
	success: sampleProcessedTime < protocolTime
	failure: sampleProcessedTime > protocolTime
