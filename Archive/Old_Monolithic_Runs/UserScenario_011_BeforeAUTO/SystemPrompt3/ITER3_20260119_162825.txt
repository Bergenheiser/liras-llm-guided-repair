SampleDelivery

agents:
	humans: Nurse, LabPersonnel
	robots: RoboticArm
locations: patientFloor, laboratory, analysisMachine
resources: sample, lockedDrawer

	1: collect(sample)
	2: insertBarcode
	3: trackSample
	4: deliver(sample, lockedDrawer)
	5: moveTo laboratory

LabPersonnel:
	1: stop
	restart if !sampleAvailable
	2: moveTo laboratory
	3: openDrawer
	4: logSample
	5: trackTime

RoboticArm:
	1: stop
	restart if  !sampleAvailable
	2: moveTo laboratory
	3: openDrawer
	4: scanBarcode
	5: loadAnalysisMachine
	6: logSample
	7: trackTime
